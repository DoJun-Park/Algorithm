#include <iostream>
#include <cstdio>
#include <fstream>


using namespace std;

void get_out_num(int sum, int multi_num){
    int out_a, out_b;
    int check = 1;

    if(sum %2 == 0) {
        out_a = (sum/2)-1;
        out_b = (sum/2)+1;
    }
    else{
        out_a = sum/2;
        out_b = sum/2 +1;
    }

    while(check){
        if(multi_num == (out_a)*(out_b)){
            check = 0;
        }
        else{
            out_a--;
            out_b++;
        }
    }
    cout << "out_a:" << out_a<<endl;
    cout << "out_b:" << out_b<<endl;

}


int main(){

    int N;
    int inp_num;
    int sum, squared_sum, multi_num;

    ifstream ifs("input.txt");
    ofstream ofs("output.txt");

    ifs >> N;

    sum = (N*(N+1))/2;
    squared_sum = (N*(N+1)*(2*N+1))/6;

    while(!ifs.eof()){
        ifs >> inp_num;
        sum -= inp_num;
        squared_sum -= inp_num*inp_num;
    }

    cout<< "sum :" <<sum<<endl;
    cout << "squared_sum:" << squared_sum<< endl;
    multi_num = ((sum*sum)-squared_sum)/2;

    get_out_num(sum, multi_num);

    ifs.close();
    return 0;

}
