#include <iostream>
#include <fstream>

using namespace std;

double A[4];
double B[4];
double C[4];
double D[4];
double len_T;
double T[4];
double S[4];
double check1, check2;

void read_data(){
    ifstream ifs("input.inp");

     for(int i=0; i<3; ++i){
        ifs >> A[i];
    }

     for(int i=0; i<3; ++i){
        ifs >> B[i];
    }

     for(int i=0; i<3; ++i){
        ifs >> C[i];
    }

     for(int i=0; i<3; ++i){
        ifs >> D[i];
    }
}

void binary_Search(double s_arr[], double e_arr[], double P[], double start, double end){
    double s[4],e[4];

    for(int i=0; i<3; ++i){
        s[i] = start* e_arr[i] + (1-start)*s_arr[i];
    }
    for(int i=0; i<3; ++i){
        e[i] = end* e_arr[i] + (1-end)*s_arr[i];
    }

    double len_s = (s[0]-P[0])*(s[0]-P[0])+(s[1]-P[1])*(s[1]-P[1])+(s[2]-P[2])*(s[2]-P[2]);
    double len_e = (e[0]-P[0])*(e[0]-P[0])+(e[1]-P[1])*(e[1]-P[1])+(e[2]-P[2])*(e[2]-P[2]);

    if(len_s == len_e){
        T[0] = (s[0] + e[0])/2;
        T[1] = (s[1] + e[1])/2;
        T[2] = (s[2] + e[2])/2;
        len_T = (T[0]-S[0])*(T[0]-S[0])+(T[1]-S[1])*(T[1]-S[1])+(T[2]-S[2])*(T[2]-S[2]);
        cout << len_T<<endl;
        return ;
    }
    else if(len_s < len_e){
        binary_Search(s_arr, e_arr, P, start,(start+end)/2);
        return ;
    }
    else{
        binary_Search(s_arr, e_arr, P,(start+end)/2, end);
        return ;
    }
}


double find_T(){
    int check = 0;

    for(int i=0; i<3; ++i){
        S[i] = C[i];
    }
    binary_Search(A,B,C,0,1);
    check1 = len_T;

    while(check < 10){
        for(int i=0; i<3; ++i){
            S[i] = T[i];
        }
        binary_Search(C,D,T,0,1);
        check2 = len_T;

        if(check1 >= check2){
            check++;
        }

        for(int i=0; i<3; ++i){
            S[i] = T[i];
        }

        check1 = check2;
        binary_Search(A,B,T,0,1);
        check2 = len_T;


        if(check1 >= check2){
            check++;
        }
        check1 = check2;
    }
    return 0;
}


void write_data(){
    ofstream ofs("output.out");
}


int main(){
    double correct;

    read_data();
    correct = find_T();
    cout<< correct<< endl;
    write_data();
    return 0;
}
