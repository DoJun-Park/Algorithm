```c++
#include <iostream>
#include <fstream>
#include <stack>

using namespace std;

int L;
int k;
int tube_len[11];


void read_data(){
    ifstream ifs("tube.inp");
    ifs >> L>>k;
    for(int i=k-1; i>=0; --i){
        ifs>>tube_len[i];
    }
    ifs.close();
}


void write_data(int min_tube_len){
    ofstream ofs("tube.out");
    ofs << min_tube_len;
    ofs.close();
}

void smallest_tube(stack <int> recursive_tube,int sum, int i){

    if(i >= k) return; //재귀는 기저조건은 항상 위에

    else{
        if((sum + tube_len[i]) == L){
//                cout<<"i2: "<<i<<endl;
//            cout << "sum2 : "<< sum <<endl;


            write_data(recursive_tube.size() +1);
//            cout <<recursive_tube.size() +1 << endl;
            exit(0);
        }

        else {
            cout<<"i: "<<i<<endl;
            cout << "sum : "<< sum <<endl;

            if(L-sum > tube_len[i]){
              sum += tube_len[i];
              recursive_tube.push(tube_len[i]);
              smallest_tube(recursive_tube, sum, i);
              return ;
            }

            if(i == (k-1)){
                if(L- sum < tube_len[i]){
                    for(int j=0; j<k; j++){
                        if(tube_len[j] == recursive_tube.top()) i = j;
                    }
                    ++i;
                    sum -= recursive_tube.top();
                    recursive_tube.pop();
                    smallest_tube(recursive_tube, sum, i);
                }
            }
        }
    }

}

void get_tube_cnt(){
    stack <int> tube;
    int i=0;
    int tube_sum =0;
    smallest_tube(tube, tube_sum, i);
    write_data(0);
}


int main(){
    read_data();
    get_tube_cnt();
    return 0;
}
```
