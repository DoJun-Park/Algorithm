#include <iostream>
#include <fstream>
#include <cstring>



using namespace std;

int n,m;
string str;
string lock_map[31][31];


void read_data(){
    ifstream ifs("lock.inp");

    ifs >> n>>m;


    for(int i=1; i<=n; ++i){
        ifs>>str;
        for(int j=1; j<=m; ++j){
            lock_map[i][j] = str.at(j-1);
         }
    }


    for(int i=1; i<=n; ++i){
        for(int j=1; j<=m; ++j){
            cout << lock_map[i][j]<<" ";
        }
        cout << endl;
    }

    cout <<lock_map[7][1]<<endl;
    ifs.close();
}




void write_data(){
    ofstream ofs("lock.out");
    ofs<<"  ";
    ofs.close();
}




int main(){
    read_data();
    write_data();
}
