#include <iostream>
#include <fstream>


using namespace std;

int k;
int vertex[53][53];
int lowest_len = 99999999;
int cycle_check = 0;


void read_data(){
    ifstream ifs("marathon.inp");

    ifs >> k;
    int s,e,len = 0;

    for(int i=0; i<k; ++i){
        ifs >> s>>e;

        while(e !=0){
            ifs >> len;
            vertex[s][e] = len;
            ifs >> e;
        }
    }

    for(int i=0; i<=k; ++i){
        for(int j =0 ; j<=k; ++j){
            if(vertex[i][j] == 0) continue;
            cout << "i: " <<i<<", j: "<<j<<" : " <<vertex[i][j];
            cout << endl;
        }
        cout << endl;
    }


    ifs.close();
}




void start(int s, int len, int first, bool check[], int cnt){
    check[s] = true;
    ++cnt;
    int temp_len = len;
    int temp_cnt = cnt;

    for(int e=1; e<=k; ++e){
        len = temp_len;
        cnt= temp_cnt;
        cout<<"**********start***********" <<endl;

        if(vertex[s][e] == 0 ) {
            cout << "111들어옴 "<<"==>start :"<<s<<", end: "<<e<<endl;
              continue;
        }

        if(check[e] == true){
            cout << "222들어옴 "<<"==>start :"<<s<<", end: "<<e<<endl;
            if(cnt >= 3 && check[e] == true){
                cout << "cnt: "<< cnt<<endl;
                if(e == first){
                    ++cycle_check;
                    len += vertex[s][e];
                    lowest_len = min(lowest_len, len);
                    cout <<"최소거리 교체 " << endl;
                    cout << "원래 최소거리: " << lowest_len<<", 교체할 최소 거리: "<< len<<endl;
                }
            }
             continue;
        }

        len += vertex[s][e];

        if(lowest_len <= len){
          cout << "lowest: "<< lowest_len<<", len: "<< len<<endl;
          cout << "333들어옴 "<< endl;
           continue;
        }


        cout <<"처음 시작 정점 :"<<first<<", start :"<<s<<", end: "<<e<<", len: "<<len<< endl;


        if(check[e] == false){
            cout<< "더 간다..."<<endl;
            cout<<"cnt: " << cnt<< endl;
             check[e] = true;
            start(e, len, first, check, cnt);
        }
    }

    return;


}


void solve(){

    for(int i=1; i<=k; ++i){
         int length=0;
         bool check[51];
         fill(check, check+51, false);
     for(int j=1; j<=k; ++j){
            if(vertex[i][j] == 0) continue;

            cout << "=======solve========="<<endl;

void write_data(){
    ofstream ofs("marathon.out");
