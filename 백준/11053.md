```c++
#include <iostream>
#include <algorithm>

using namespace std;

int sequence[1001];
int dp[1001];
int cnt, ans;

int main(){


    cin >> cnt;

    for(int i=0; i<cnt; i++){
        cin >> sequence[i];
    }

    for(int i=0; i<cnt; i++){
        dp[i]=1;
        for(int j=0; j<i; j++){
            if(sequence[i] > sequence[j] && dp[j]+1 >dp[i]){
                dp[i] = dp[j]+1;
            }
        }
    }

    for(int i=0; i< cnt; i++){
        ans = max(ans, dp[i]);
    }

    cout << ans;
    return 0;
}
```
