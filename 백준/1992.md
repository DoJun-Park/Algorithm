#include <iostream>
#include <stdio.h>
using namespace std;


int array[65][65];


void Quad(int sx, int sy, int ex, int ey){

    int check = array[sx][sy];

    for(int i=sx; i<ex; ++i){
        for(int j=sy; j<ey; ++j){
            if(check == 0 && array[i][j] == 1){
                check = 2;
            }
            else if(check == 1 && array[i][j] == 0){
                check = 2;
            }
            if(check == 2){
                cout<<"(";
                Quad(sx, sy, (sx+ex)/2, (sy+ey)/2);
                Quad(sx, (sy+ey)/2, (sx+ex)/2,ey);
                Quad((sx+ex)/2, sy, ex,(sy+ey)/2);
                Quad((sx+ex)/2, (sy+ey)/2, ex,ey);
                cout<< ")";
                return ;
            }
        }
    }
    cout << array[sx][sy];
    return ;
}

int main(){

    int N;
    cin >> N;

    for(int i=0; i<N; ++i){
        for(int j=0; j<N; ++j){
            scanf("%1d", &array[i][j]);
//            cin >> array[i][j];
        }
    }

    Quad(0,0,N,N);
    return 0;
}
